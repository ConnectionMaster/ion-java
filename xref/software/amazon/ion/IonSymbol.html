15  package software.amazon.ion;
16  
17  /**
18   * An Ion <code>symbol</code> value.
19   * <p>
20   * <b>WARNING:</b> This interface should not be implemented or extended by
21   * code outside of this library.
22   */
23  public interface IonSymbol
24      extends IonText
25  {
26      /**
27       * Gets the text content of this symbol.
28       *
29       * @return the text of the symbol, or <code>null</code> if this is
30       * <code>null.symbol</code>.
31       *
32       * @throws UnknownSymbolException if this symbol has unknown text.
33       */
34      public String stringValue()
35          throws UnknownSymbolException;
36  
37  
38      /**
39       * Returns this value as a symbol token (text + ID).
40       *
41       * @return null if {@link #isNullValue()}
42       *
43       */
44      public SymbolToken symbolValue();
45  
46  
47      /**
48       * Changes the value of this element.
49       *
50       * @param value the new value of this symbol;
51       * may be <code>null</code> to make this <code>null.symbol</code>.
52       *
53       */
54      public void setValue(String value);
55  
56      public IonSymbol clone()
57          throws UnknownSymbolException;
58  }
