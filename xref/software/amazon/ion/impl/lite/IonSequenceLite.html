350     @SuppressWarnings("unchecked")
351     public <T> T[] toArray(T[] a)
352     {
353         int size = get_child_count();
354         if (a.length < size)
355         {
356             // TODO JDK 1.6 this could use Arrays.copyOf
357             Class<?> type = a.getClass().getComponentType();
358             // generates unchecked warning
359             a = (T[]) Array.newInstance(type, size);
360         }
361         if (size > 0) {
362             System.arraycopy(_children, 0, a, 0, size);
363         }
364         if (size < a.length) {
365             // A surprising bit of spec.
366             // this is required even with a 0 entries
367             a[size] = null;
368         }
369         return a;
370     }
