15  package software.amazon.ion.impl.lite;
16  
17  import java.io.IOException;
18  import software.amazon.ion.IonBool;
19  import software.amazon.ion.IonType;
20  import software.amazon.ion.IonWriter;
21  import software.amazon.ion.NullValueException;
22  import software.amazon.ion.ValueVisitor;
23  
24  final class IonBoolLite
25      extends IonValueLite
26      implements IonBool
27  {
28      private static final int HASH_SIGNATURE =
29          IonType.BOOL.toString().hashCode();
30  
31      /**
32       * Optimizes out a function call for a const result
33       */
34      protected static final int TRUE_HASH
35              = HASH_SIGNATURE ^ (16777619 * Boolean.TRUE.hashCode());
36  
37      /**
38       * Optimizes out a function call for a const result
39       */
40      protected static final int FALSE_HASH
41              = HASH_SIGNATURE ^ (16777619 * Boolean.FALSE.hashCode());
42  
43      /**
44       * Constructs a null bool value.
45       */
46      IonBoolLite(ContainerlessContext context, boolean isNull)
47      {
48          super(context, isNull);
49      }
50  
51      IonBoolLiteref="../../../../../software/amazon/ion/impl/lite/IonBoolLite.html#IonBoolLite">IonBoolLite(IonBoolLite existing, IonContext context)
52      {
53          super(existing, context);
54      }
55  
56      @Override
57      IonBoolLite clone(IonContext context)
58      {
59          return new IonBoolLite(this, context);
60      }
61  
62      @Override
63      public IonBoolLite clone()
64      {
65          return clone(ContainerlessContext.wrap(getSystem()));
66      }
67  
68      @Override
69      public IonType getType()
70      {
71          return IonType.BOOL;
72      }
73  
74      @Override
75      int hashCode(SymbolTableProvider symbolTableProvider)
76      {
77          int result = HASH_SIGNATURE;
78  
79          if (!isNullValue())
80          {
81              result = booleanValue() ? TRUE_HASH : FALSE_HASH;
82          }
83  
84          return hashTypeAnnotations(result, symbolTableProvider);
85      }
86  
87      public boolean booleanValue()
88          throws NullValueException
89      {
90          validateThisNotNull();
91          return this._isBoolTrue();
92      }
93  
94      public void setValue(boolean b)
95      {
96          // the called setValue will check if this is locked
97          setValue(Boolean.valueOf(b));
98      }
99  
100     public void setValue(Boolean b)
101     {
102         checkForLock();
103         if (b == null) {
104             _isBoolTrue(false);
105             _isNullValue(true);
106         }
107         else {
108             _isBoolTrue(b.booleanValue());
109             _isNullValue(false);
110         }
111     }
112 
113     @Override
114     final void writeBodyTo(IonWriter writer, SymbolTableProvider symbolTableProvider)
115         throws IOException
116     {
117         if (isNullValue())
118         {
119             writer.writeNull(IonType.BOOL);
120         }
121         else
122         {
123             writer.writeBool(_isBoolTrue());
124         }
125     }
126 
127     @Override
128     public void accept(ValueVisitor visitor)
129         throws Exception
130     {
131         visitor.visit(this);
132     }
133 }
