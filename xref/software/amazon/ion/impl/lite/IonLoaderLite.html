34  final class IonLoaderLite
35      implements IonLoader
36  {
37      private final IonSystemLite _system;
38  
39      /** Not null. */
40      private final IonCatalog    _catalog;
41  
42      private final PrivateLocalSymbolTableFactory _lstFactory;
43  
44      /**
45       * @param system must not be null.
46       * @param catalog must not be null.
47       */
48      public IonLoaderLite(IonSystemLite system, IonCatalog catalog)
49      {
50          assert system != null;
51          assert catalog != null;
52  
53          _system = system;
54          _catalog = catalog;
55          _lstFactory = system.getLstFactory();
56      }
57  
58      public IonSystem getSystem()
59      {
60          return _system;
61      }
62  
63      public IonCatalog getCatalog()
64      {
65          return _catalog;
66      }
67  
68  
69      /**
70       * This doesn't wrap IOException because some callers need to propagate it.
71       *
72       * @return a new datagram; not null.
73       */
74      private IonDatagramLite load_helper(IonReader reader)
75      throws IOException
76      {
77          IonDatagramLiteagramLite.html#IonDatagramLite">IonDatagramLite datagram = new IonDatagramLite(_system, _catalog);
78          IonWriter writer = PrivateIonWriterFactory.makeWriter(datagram);
79          writer.writeValues(reader);
80          return datagram;
81      }
82  
83      public IonDatagram load(File ionFile) throws IonException, IOException
84      {
85          InputStream ionData = new FileInputStream(ionFile);
86          try
87          {
88              IonDatagram datagram = load(ionData);
89              return datagram;
90          }
91          finally
92          {
93              ionData.close();
94          }
95      }
96  
97      public IonDatagram load(String ionText) throws IonException
98      {
99          try {
100             IonReader reader = makeReader(_catalog, ionText, _lstFactory);
101             IonDatagramLite datagram = load_helper(reader);
102             return datagram;
103         }
104         catch (IOException e) {
105             throw new IonException(e);
106         }
107     }
108 
109     public IonDatagram load(Reader ionText) throws IonException, IOException
110     {
111         try {
112             IonReader reader = makeReader(_catalog, ionText, _lstFactory);
113             IonDatagramLite datagram = load_helper(reader);
114             return datagram;
115         }
116         catch (IonException e) {
117             IOException io = e.causeOfType(IOException.class);
118             if (io != null) throw io;
119             throw e;
120         }
121     }
122 
123     public IonDatagram load(byte[] ionData) throws IonException
124     {
125         try {
126             IonReader reader = makeReader(_catalog, ionData, 0, ionData.length, _lstFactory);
127             IonDatagramLite datagram = load_helper(reader);
128             return datagram;
129         }
130         catch (IOException e) {
131             throw new IonException(e);
132         }
133     }

148     }
