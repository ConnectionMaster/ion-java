90       * May be used by subclasses to reuse _flag bits for purposes specific
91       * to that subclass. It is important that only flag bits not currently
92       * used by that subclass are chosen; otherwise important data may be
93       * overwritten. NOTE: only the lower 8 bits may be used, because the
94       * upper 24 are reserved for the element ID.
95       * @param metadata the metadata to set.

691     protected int hashTypeAnnotations(final int original, SymbolTableProvider symbolTableProvider)
692     {
693         final SymbolToken[] tokens = getTypeAnnotationSymbols(symbolTableProvider);
694         if (tokens.length == 0)
695         {
696             return original;
697         }
698 
699         final int sidHashSalt   = 127;      // prime to salt sid of annotation
700         final int textHashSalt  = 31;       // prime to salt text of annotation
701         final int prime = 8191;
702         int result = original ^ TYPE_ANNOTATION_HASH_SIGNATURE;
703 
704         result = prime * original + tokens.length;
705 
706         for (final SymbolToken token : tokens)
707         {
708             String text = token.getText();
709 
710             int tokenHashCode = text == null
711                 ? token.getSid()  * sidHashSalt
712                 : text.hashCode() * textHashSalt;
713 
714             // mixing to account for small text and sid deltas
715             tokenHashCode ^= (tokenHashCode << 19) ^ (tokenHashCode >> 13);
716 
717             result = prime * result + tokenHashCode;
718 
719             // mixing at each step to make the hash code order-dependent
720             result ^= (result << 25) ^ (result >> 7);
721         }
722 
723         return result;
724     }
