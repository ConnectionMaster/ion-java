78      {
79          if (_eof) return false;
80          if (_next != null) return true;

83

99                      }

102                         String name = sym.stringValue();

184     private final Span currentSpanImpl()
185     {
186         if (this._curr == null) {
187             throw new IllegalStateException("Reader has no current value");
188         }
189 
190         TreeSpan span = new TreeSpan();
191         span._value = this._curr;
192 
193         return span;
194     }

197     private void hoistImpl(Span span)
198     {
199         if (span instanceof TreeSpan) {
200             TreeSpan treeSpan = (TreeSpan)span;
201             this.re_init(treeSpan._value, /* hoisted */ true);
202         }
203         else {
204             // TODO custom exception
205             throw new IllegalArgumentException("Span not appropriate for this reader");
206         }
207     }
