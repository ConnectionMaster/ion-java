5   /**
6    * Indicates the smallest-possible Java type of an Ion {@code int} value.
7    */
8   public enum IntegerSize
9   {
10      /**
11       * Fits in the Java {@code int} primitive (four bytes).
12       * The value can be retrieved through methods like {@link IonReader#intValue()}
13       * or {@link IonInt#intValue()} without data loss.
14       */
15      INT,
16  
17      /**
18       * Fits in the Java {@code int} primitive (eight bytes).
19       * The value can be retrieved through methods like {@link IonReader#longValue()}
20       * or {@link IonInt#longValue()} without data loss.
21       */
22      LONG,
23  
24      /**
25       * Larger than eight bytes. This value can be retrieved through methods like
26       * {@link IonReader#bigIntegerValue()} or {@link IonInt#bigIntegerValue()}
27       * without data loss.
28       */
29      BIG_INTEGER,
30  
31  }
